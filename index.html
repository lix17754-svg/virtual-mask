        let currentQuestionIndex = 0;

        let answers = {};


        function getElement(id) {

            return document.getElementById(id);

        }


        function showScreen(screenId) {

            ['home-screen', 'question-screen', 'result-screen', 'contact-screen'].forEach(id => {

                const el = getElement(id);

                if (el) {

                    el.classList.add('hidden');

                    el.classList.remove('fade-in');

                }

            });

            const screen = getElement(screenId);

            screen.classList.remove('hidden');

            // Trigger reflow to restart animation

            void screen.offsetWidth;

            screen.classList.add('fade-in');

        }


        function startTest() {

            currentQuestionIndex = 0;

            answers = {};

            renderQuestion();

            showScreen('question-screen');

        }


        function renderQuestion() {

            const q = questions[currentQuestionIndex];

            

            getElement('progress-text').innerText = `${currentQuestionIndex + 1} / ${questions.length}`;

            getElement('progress-bar').style.width = `${((currentQuestionIndex + 1) / questions.length) * 100}%`;

            

            // 控制“上一题”按钮的显示/隐藏

            const prevBtn = getElement('prev-btn');

            if (currentQuestionIndex > 0) {

                prevBtn.classList.remove('hidden');

                prevBtn.classList.add('inline-flex');

            } else {

                prevBtn.classList.add('hidden');

                prevBtn.classList.remove('inline-flex');

            }


            getElement('question-title').innerText = q.text;

            

            const optionsContainer = getElement('options-container');

            optionsContainer.innerHTML = '';

            

            q.options.forEach((opt, index) => {

                const btn = document.createElement('button');

                // 如果用户之前选过这个选项，给它加上一个选中的高亮样式

                const isSelected = answers[currentQuestionIndex] === opt.value;

                const activeClass = isSelected ? 'border-indigo-400 bg-indigo-900/30 ring-1 ring-indigo-500' : 'border-gray-600 bg-gray-800/50 hover:border-indigo-400';

                

                btn.className = `option-btn w-full text-left p-4 rounded-lg border focus:outline-none focus:ring-2 focus:ring-indigo-500 mb-3 transition-all duration-200 ${activeClass}`;

                btn.innerHTML = `<span class="inline-block w-6 font-bold text-indigo-400">${String.fromCharCode(65 + index)}.</span> <span class="text-gray-200">${opt.text}</span>`;

                

                // 将 index 传给 selectOption，用于视觉高亮

                btn.onclick = () => selectOption(opt.value, index);

                optionsContainer.appendChild(btn);

            });

        }


        function prevQuestion() {

            if (currentQuestionIndex > 0) {

                currentQuestionIndex--;

                renderQuestion();

            }

        }


        function selectOption(value, index) {

            answers[currentQuestionIndex] = value;

            

            // 点击时的视觉反馈

            const buttons = document.querySelectorAll('#options-container button');

            buttons.forEach(btn => {

                btn.className = "option-btn w-full text-left p-4 rounded-lg border focus:outline-none focus:ring-2 focus:ring-indigo-500 mb-3 transition-all duration-200 border-gray-600 bg-gray-800/50 hover:border-indigo-400";

            });

            const clickedBtn = buttons[index];

            clickedBtn.className = "option-btn w-full text-left p-4 rounded-lg border focus:outline-none focus:ring-2 focus:ring-indigo-500 mb-3 transition-all duration-200 border-indigo-400 bg-indigo-900/30 ring-1 ring-indigo-500 transform scale-[1.02]";


            if (currentQuestionIndex < questions.length - 1) {

                setTimeout(() => {

                    currentQuestionIndex++;

                    renderQuestion();

                }, 250); // 稍微延长一点时间，让用户看清自己选了什么

            } else {

                setTimeout(() => {

                    calculateResult();

                }, 250);

            }

        }


        function calculateResult() {

            // 统计所有 6 个维度的得分

            let scores = {

                boundary: 0,     // 边界感+反击性 (拒绝越界)

                observation: 0,  // 风险意识 (先观察打听)

                obedience: 0,    // 服从度 (先忍、先配合)

                resource: 0,     // 公平感/资源现实 (关注钱/机会)

                social: 0,       // 社交能量 (愿意社交)

                rebellion: 0,    // 反击性/风险意识 (当场追责)

                fairness: 0,     // 强公平感

                risk: 0          // 强风险意识/留痕

            };


            for (let i = 0; i < questions.length; i++) {

                if (answers[i]) {

                    scores[answers[i]]++;

                }

            }


            // 根据高分组合匹配角色

            let personaKey = "Earth_Ice"; // 默认沙僧兜底


            // 提取最高的两个维度

            let sortedDims = Object.keys(scores).sort((a, b) => scores[b] - scores[a]);

            let top1 = sortedDims[0];

            let top2 = sortedDims[1];

            

            // 为了方便判断，构建一个组合字符串 (包含前两个维度)

            let combo = [top1, top2].sort().join('_');

            

            // 匹配逻辑 (基于用户提供的图表)

            if (scores.rebellion > 1 && scores.fairness > 1) {

                personaKey = "Fire_Wind"; // 孙悟空：反击性高 + 公平感高

            } else if (scores.boundary > 1 && (scores.fairness > 1 || scores.risk > 1)) {

                personaKey = "Ice_Earth"; // 唐僧：边界感高 + 原则感强

            } else if (scores.resource > 1 && scores.boundary > 0) {

                personaKey = "Wind_Earth"; // 猪八戒：回报意识高 + 抗消耗高

            } else if (scores.obedience > 2) {

                personaKey = "Earth_Ice"; // 沙僧：服从度高 + 稳定性高

            } else if (scores.obedience > 1 && scores.social === 0) {

                personaKey = "Earth_Earth"; // 白龙马：服从度高 + 表达低

            } else if (scores.social > 2) {

                personaKey = "Wind_Wind"; // 观音：社交能量高

            } else if (scores.risk > 2 || (scores.risk > 1 && scores.observation > 1)) {

                personaKey = "Ice_Ice"; // 太上老君：风险意识高 + 留痕意识高

            } else if (scores.boundary > 2 || (scores.boundary > 1 && scores.rebellion > 0)) {

                personaKey = "Fire_Ice"; // 二郎神：边界感高 + 反击性高

            } else if (scores.resource > 2 || (scores.resource > 1 && scores.fairness > 1)) {

                personaKey = "Fire_Fire"; // 牛魔王：公平感高 + 资源意识高

            } else if (scores.observation > 2) {

                personaKey = "Wind_Ice"; // 白骨精：风险意识高 + 观察性高

            }


            const pData = personasData[personaKey] || personasData["Earth_Ice"];

            

            // 渲染数据

            getElement('result-title').innerText = pData.title;

            getElement('result-image').src = pData.image; // 设置角色头像

            getElement('result-quote').innerText = `"${pData.quote}"`;

            getElement('result-summary').innerText = pData.summary;

            getElement('result-explanation').innerText = pData.explanation;

            

            // 渲染列表 (表现)

            const behaviorsHtml = pData.behaviors.map(b => `<li>${b}</li>`).join('');

            getElement('result-behaviors').innerHTML = behaviorsHtml;

            

            // 渲染列表 (优势)

            const prosHtml = pData.pros.map(p => `<li>${p}</li>`).join('');

            getElement('result-pros').innerHTML = prosHtml;

            

            // 渲染列表 (风险)

            const consHtml = pData.cons.map(c => `<li>${c}</li>`).join('');

            getElement('result-cons').innerHTML = consHtml;

            

            // 渲染建议

            getElement('result-advice').innerText = pData.advice;

            

            showScreen('result-screen');

        }

        function shareResult() {

            const title = getElement('result-title').innerText;

            const quote = getElement('result-quote').innerText;

            const shareText = `我在【撕开你的职场面具】测试中测出了：\n${title}\n${quote}\n\n快来测测你的真实职场底色吧！`;

            

            if (navigator.share) {

                navigator.share({

                    title: '撕开你的职场面具',

                    text: shareText,

                    url: window.location.href

                }).catch(console.error);

            } else {

                if (navigator.clipboard && window.isSecureContext) {

                    navigator.clipboard.writeText(shareText).then(() => {

                        alert("结果已复制到剪贴板，快去分享给朋友吧！");

                    }).catch(() => {

                        alert("分享功能暂不可用，请截图分享哦~");

                    });

                } else {

                    // Fallback for non-HTTPS or missing clipboard API

                    const textArea = document.createElement("textarea");

                    textArea.value = shareText;

                    textArea.style.position = "fixed";

                    textArea.style.left = "-999999px";

                    textArea.style.top = "-999999px";

                    document.body.appendChild(textArea);

                    textArea.focus();

                    textArea.select();

                    try {

                        document.execCommand('copy');

                        alert("结果已复制到剪贴板，快去分享给朋友吧！");

                    } catch (err) {

                        alert("分享功能暂不可用，请截图分享哦~");

                    }

                    textArea.remove();

                }

            }

        }
